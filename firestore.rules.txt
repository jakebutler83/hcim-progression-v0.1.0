rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isCurrentMember(groupId) {
      return signedIn()
        && request.auth.uid in
          get(/databases/$(database)/documents/groups/$(groupId)).data.memberUids;
    }

    function isMemberAfterWrite(groupId) {
      return signedIn()
        && request.auth.uid in
          getAfter(/databases/$(database)/documents/groups/$(groupId)).data.memberUids;
    }

    match /users/{userId} {
      allow read, write: if signedIn()
                         && request.auth.uid == userId;
    }

    match /groupInvites/{inviteCode} {
      allow get: if signedIn();
      allow list: if false;

      allow create: if signedIn()
                    && request.resource.data.ownerUid == request.auth.uid;

      allow update, delete: if signedIn()
                            && resource.data.ownerUid == request.auth.uid;
    }

    match /groups/{groupId} {
      allow create: if signedIn()
                    && request.resource.data.ownerUid == request.auth.uid
                    && request.resource.data.memberUids == [request.auth.uid];

      // Temporary for the join flow: a signed-in user may fetch one group document
      // after resolving a private invite code. Collection listing stays blocked.
      allow get: if signedIn();
      allow list: if false;

      allow update: if signedIn() && (
        isCurrentMember(groupId)
        || (
          !(request.auth.uid in resource.data.memberUids)
          && request.auth.uid in request.resource.data.memberUids
          && request.resource.data.memberUids.size()
             == resource.data.memberUids.size() + 1
          && request.resource.data.memberCount
             == resource.data.memberCount + 1
          && request.resource.data.memberCount
             <= resource.data.groupSize
          && request.resource.data
               .diff(resource.data)
               .affectedKeys()
               .hasOnly([
                 'memberUids',
                 'memberCount',
                 'updatedAt'
               ])
        )
      );

      allow delete: if signedIn()
                    && request.auth.uid == resource.data.ownerUid;

      match /members/{memberUid} {
        allow read: if isCurrentMember(groupId);

        allow create: if signedIn()
                      && request.auth.uid == memberUid
                      && isMemberAfterWrite(groupId);

        allow update, delete: if isCurrentMember(groupId);
      }

      match /progress/{progressId} {
        allow read: if isCurrentMember(groupId);
        allow create, update: if isMemberAfterWrite(groupId);
        allow delete: if isCurrentMember(groupId);
      }

      match /{subcollection}/{documentId} {
        allow read: if isCurrentMember(groupId);
        allow write: if isMemberAfterWrite(groupId);
      }
    }

    match /hcimTrackers/{trackerId} {
      allow read, write: if signedIn();
    }
  }
}
